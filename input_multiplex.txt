# input_multiplex.txt
GRID 16 16

RESERVOIR R1 3 1 URINE
RESERVOIR R2 8 1 PLASMA
RESERVOIR R3 13 1 SERUM
RESERVOIR R4 1 6 GLUCOSE_OXIDASE
RESERVOIR R5 16 10 LACTATE_OXIDASE

OUTPUT O1 16 16
WASTE  W1 1 1

# Concurrent initial reagent processing
MIX URINE GLUCOSE_OXIDASE -> GLU_URINE
MIX PLASMA GLUCOSE_OXIDASE -> GLU_PLASMA
MIX SERUM LACTATE_OXIDASE -> LAC_SERUM

# Downstream analysis mixtures
MIX GLU_URINE GLU_PLASMA -> COMBINED_GLUCOSES
MIX LAC_SERUM COMBINED_GLUCOSES -> FINAL_PANEL
