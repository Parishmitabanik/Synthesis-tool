# input_stress_test.txt
GRID 20 30

RESERVOIR SM1 2 1 SAMPLE_1
RESERVOIR SM2 6 1 SAMPLE_2
RESERVOIR SM3 12 1 SAMPLE_3
RESERVOIR SM4 18 1 SAMPLE_4
RESERVOIR BUF 1 15 BUFFER
RESERVOIR REA 20 15 REAGENT

OUTPUT O1 10 30
WASTE  W1 1 1

# Massive parallel cross-routing
MIX SAMPLE_1 BUFFER -> MIX_1
MIX SAMPLE_2 REAGENT -> MIX_2
MIX SAMPLE_3 BUFFER -> MIX_3
MIX SAMPLE_4 REAGENT -> MIX_4

# Consolidation stages
MIX MIX_1 MIX_2 -> BATCH_A
MIX MIX_3 MIX_4 -> BATCH_B
MIX BATCH_A BATCH_B -> HIGH_THROUGHPUT_FINAL
